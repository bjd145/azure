| mv-expand  agentPoolProfiles
| extend 
    nodeName = tostring(agentPoolProfiles.name),
    nodeImageVersion = tostring(agentPoolProfiles.nodeImageVersion)
| project resourceGroup, name, nodeName, nodeImageVersion
